Seletor Pseudoclasse CSS

   /* C S S */
   :link { color: blue; } /* Links não visitados serão na cor azul */ 
   :focus { color: white; } /* Elementos que ganham o foco serão na cor branca */ 
   a.externo:visited { color: red; } /* Links externos que já tenham sido visitados serão na cor vermelha */ 
   a:focus:visited { color: green; } /* É válido definir mais de uma pseudoclasse */
   a:hover { color: gray;} /* ao passar o mause por cima muda a cor das letras para cinza  */
   p:hover {background-color: green; color: yellow;}

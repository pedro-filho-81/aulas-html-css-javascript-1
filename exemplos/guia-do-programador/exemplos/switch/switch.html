Switch JavaScript

Exemplo JavaScript

   let estadoDoSul = prompt("Digite a sigla de um Estado da região sul: ");

   switch(estadoDoSul) {
      case "RS":
      case "Rs":
      case "rs":
         alert("Rio Grande Do Sul");
         break;

      case "SC":
      case "Sc":
      case "sc":
         alert("Santa Catarina");
         break;

      case "PR": 
      case "Pr": 
      case "pr": 
         alert("Paraná");
         break;

      default: 
         alert("Errado! Esse não é um Estado do Região Sul.");
   }
